public class Interface {
    public static void main(String[] args) {

        Animal a = new Dog();
        Animal b = new Cat();

        a.sound();
        b.sound();
    }
}
interface Animal {
    void sound();
}

class Dog implements Animal {
    public void sound() {
        System.out.println("Bow Bow");
    }
}

class Cat implements Animal {
    public void sound() {
        System.out.println("Meow Meow");
    }
}
